# NeoROS
ROS package for Neocobot's arm.

# Build
NeoROS package depends on C++11.

So, Please add this command in your CMakeLists.txt that you can find in ROS_workspace/src.

set(CMAKE_CXX_FLAGS "-std=c++11 ${CMAKE_CXX_FLAGS}")
